#include <iostream>

using namespace std;

int main()
{
	const int size = 10;
	int ar[size], x;
	for (int i = 0; i < size; i++)
		cin >> ar[i];
	cin >> x;
	int l = 0, r = size, mid, i = -1;
	while (l < r)
	{
		mid = (l + r) / 2;
		if (ar[mid] == x) {
			i = mid;
			break;
		}
		if (x > ar[mid]) {
			l = mid+1;
		}
		else {
			r = mid;
		}
	}
	cout << "Otvet: " << i << endl;
	cin.get(); cin.get();
}
