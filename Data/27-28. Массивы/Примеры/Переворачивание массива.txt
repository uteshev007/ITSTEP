#include<iostream>
#include<time.h>
using namespace std;

int main() {
	const int size = 10;
	int mas[size];

	for (int i = 0; i < size; i++) {
		mas[i] = i+1;
	}
	for (int i = 0; i < size; i++) {
		cout << mas[i] << " ";
	}

	for (int i = 0; i < size/2; i++) {
		swap(mas[i], mas[size - 1 - i]);
	}
	cout << endl;
	for (int i = 0; i < size; i++) {
		cout << mas[i] << " ";
	}
	cin.get();
}
