#include <iostream>

using namespace std;

enum {
  // коды символов, клавиатурных кнопок - getch()
  ENTER = 13,
  ESC = 27,
  // коды (двойные) для кнопок управления курсором
  //  двухкратный вызов getch()
  CURSOR1 = 0,
  CURSOR2 = 224,
  UP = 72,
  DOWN = 80,
  RIGHT = 77,
  LEFT = 75
};

// Установка позиции курсора (каретки) консоли 
//  на позицию y - номер строки, x - номер столбца
void SetPos(int x, int y)
{
  COORD cd;
  cd.X = x;
  cd.Y = y;
  SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), cd);
}
