// макро-функция, макрос для генерации случ. чисел
#define RND(min, max)  (rand()%((max)-(min)+1)+(min))
// RND(2+3*4, 5-8/5) * 10 ==>
//  (rand()%((5-8/5) - (2+3*4) + 1) + (2+3*4)) * 10
/*
33. **Написать функцию, которая принимает указатель
на массив и количество элементов.
Функция сжимает массив, удаляя из него элементы равные 0.
Возвращает указатель на новый массив, не содержащий 0.
*/
int * DeleteZero(int* mas, int size, int& newsize) {
  newsize = 0;
  for (int i = 0; i < size; i++) {
    if (mas[i] != 0) {
      newsize++;
    }
  }
  // создание нового массива
  int* mas2 = new int[newsize];
  // копирование в mas2
  int j = 0;
  for (int i = 0; i < size; i++) {
    if (mas[i] != 0) {
      mas2[j] = mas[i];
      j++;
    }
  }
  // возвращаем указатель на новый массив без нулей
  return mas2;
}
