/*Напишите функцию, которая принимает одномерный массив целых чисел и
возвращает сумму всех чисел*/

int summa(int arr[], int size) {
  int sum = 0;
  for (int i = 0; i < size; i++) {
    sum += arr[i];
}
  return sum;
}

int main()
{
  setlocale(LC_ALL, "Rus");
  srand(time(0));
  const int size = 10;
  int mas[size];

  for (int i = 0; i < size; i++) {
    mas[i] = rand() % 10 + 1;
    cout << mas[i] << " ";
  }
  cout << endl;
  int sum = 0;
  for (int i = 0; i < size; i++) {
    sum += mas[i];
  }
  cout << "Summa = " << sum << endl;
  cout << "Summa = " << summa(mas, size) << endl;
  int mas2[size * 2] = { 5, 3, 19, 7, 8, 6 };
  cout << "Summa 2 = " << summa(mas2, size * 2) << endl;
  cin.get(); cin.get();
}
