// Ф-ция для вычисления куба числа
//  принимает один аргумент - a
//  не возвращает явного значения
// Тип функции - void - ф-ция не возвращает явного значения
void Cub1(int a)
{
  cout << "a**3 = " << a * a * a << endl;
  cout << "a**3 = " << pow(a,3) << endl;
  // return - оператор для выхода из ф-ции
  return; // выход из ф-ции 
  // В этой ф-ции (типа void) return не обязателен
}

// Функция вычисления куба числа
//  принимает один агрумент, параметр - <a>
//  возвращает явное значение - куб числа из <a>
// Тип функции - int - возвращает явное значение типа int
int Cub2(int a)
{
  int result = a * a * a; // куб числа
  // return - оператор для выхода из ф-ции
  return result; // возврат значения из функции
  // оператор return в этой ф-ции ОБЯЗАТЕЛЕН
}

int main()
{
  setlocale(LC_ALL, "Rus");
  srand(time(0));
  int num;
  cout << "Введите цело число: ";
  cin >> num;
  // 1 вызов ф-ции Cub1()
  Cub1(num); // в ф-цию Cub1() передаем аргумент num
  // 2 вызов ф-ции Cub2() с сохранением результата в переменную
  int res = Cub2(num); // сохраняем результат вызова функции
  cout << num << "**3 = " << res << endl;
  // 2.1 вызов ф-ции Cub2() без сохранения результата
  cout << num << "**3 = " << Cub2(num) << endl;

  cin.get(); cin.get();
}
