1 и на себя(2, 5, 7, 11 и т.д.).*/

// Ф-ция проверки числа на простоту
bool CheckPrime(int a)
{
  for (int i = 2; i < a; i++) {
    if (a % i == 0) {
      return false;
    }
  }
  return true;
}

  if (ans == true) {

  }
  else {

  }
