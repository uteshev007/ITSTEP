#include <iostream>

using namespace std;

double uravnenie(double a1, double b1, double c1, double x1)
{
	return a1 * x1 * x1 + b1 * x1 + c1;
}

int main()
{
	double a, b, c, x, result;
	cout << "Vvedite koefficienty uravneniya" << endl;
	cin >> a >> b >> c;
	cout << "Vvedite x" << endl;
	cin >> x;
	result = uravnenie(2, 3, 4, 2);
	cout<<result<<endl;
	cin.get(); cin.get();
}
