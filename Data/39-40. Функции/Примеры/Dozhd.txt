#include <iostream>

using namespace std;

bool IdetLiNaUliceDozhd(bool mokroeLiOkno, bool estLiTuchi) {
	if (mokroeLiOkno == true && estLiTuchi == true)
		return true;
	else
		return false;
}

int main()
{
	bool mokroeOkno, nalichieTuch, idetLiDozhd;
	char otvet;
	cout << "Mokroe li okno? Y/N" << endl; 
	cin >> otvet;
	if (otvet == 'Y')
		mokroeOkno = true;
	else
		mokroeOkno = false;
	cout << "Est li na ulice tuchi? Y/N" << endl;
	cin >> otvet;
	if (otvet == 'Y')
		nalichieTuch = true;
	else
		nalichieTuch = false;
	idetLiDozhd = IdetLiNaUliceDozhd(mokroeOkno, nalichieTuch);
	if (idetLiDozhd == true)
		cout << "Na ulice idet dozhd" << endl;
	else
		cout << "Na ulice ne idet dozhd" << endl;
	cin.get(); cin.get();
}
