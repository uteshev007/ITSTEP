#include <iostream>
#include <time.h>  // time()
using namespace std;

/*
 Циклы, массивы
  Сортировка
*/
#define SIZE 10
int main()
{
  setlocale(LC_ALL, "Rus");
  srand(time(0));
  // константная переменная для массива
  const int Size = 10;
  int mas[Size]; // целочисленный массив на Size элементов
  for (int i = 0; i < Size; i++) {
    mas[i] = rand() % 9 + 1; // [1;9] - рандомное наполнение массива
    cout << mas[i] << " "; // вывод на экран через пробел
  }
  cout << endl;
  /*// сортировка "пузырьком", "пузырьковая" сортировка
  for (int i = 0; i < Size; i++) // цикл по раундам сортировки
  {                              //  - одинаковый для обменных типов сортировки
    for (int j = Size - 1; j > 0; j--) { // цикл "пузырьковой" сортировки
      if (mas[j] < mas[j - 1]) {
        swap(mas[j], mas[j - 1]); // обмен двух соседних элементов массива
      }
    }
  }*/
  // Простая обменная сортировка - сортировка "кирпичем"
  for (int i = 0; i < Size; i++) { // цикл по раундам сортировки
    for (int j = 0; j < Size - 1; j++) { // цикл простой обменная сортировка
      //if (mas[j] > mas[j + 1]) { // по возрастанию
      if (mas[j] < mas[j + 1]) { // по убыванию
        swap(mas[j], mas[j + 1]); // обмен
      }
    }
  }

  // вывод результата сортировки
  cout << "Отсортированный массив:\n";
  for (int i = 0; i < Size; i++) {
    cout << mas[i] << " ";
  }
  cout << endl;

  cin.get(); cin.get();
}
