#include<iostream>
using namespace std;

int main() {
	const int size = 10;
	int mas[size] = {9,8,7,6,5,4,3,2,1,0};
	int min = 0;

	for (int i = 0; i < size - 1; i++) {
		min = i;
		for (int j = i; j < size; j++) {
			if (mas[min] > mas[j]) {
				min = j;
			}
		}
		swap(mas[min], mas[i]);
	}

	for (int i = 0; i < size; i++) {
		cout << mas[i] << " ";
	}
	cin.get();
}
