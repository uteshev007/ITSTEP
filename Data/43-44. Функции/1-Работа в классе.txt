void InitMas2D(int mas[ROW][COL], int row, int col,
  int min, int max)
{
  for (int i = 0; i < row; i++)
  {
    for (int j = 0; j < col; j++)
    {
      mas[i][j] = rand() % (max - min + 1) + min;
    }
  }
}

void PrintMas2D(int mas[ROW][COL], int row, int col)
{
  for (int i = 0; i < row; i++)
  {
    cout << "За " << i + 1 << " год: ";
    for (int j = 0; j < col; j++)
    {
      cout << setw(3) << mas[i][j] << " ";
    }
    cout << setw(3) << "|" << SumMas2D(mas, i, col) << " " << endl;
  }
}

int SumMas2D(int mas[ROW][COL], int row, int col)
{
  int sum = 0;
  //for (int i = 0; i < row; i++)
  {
    for (int j = 0; j < col; j++)
    {
      sum += mas[row][j];
    }
  }
  return sum;
}

void GrafMas2D(int mas[ROW][COL], int row, int col, int height)
{
  int sum[ROW];
  int max = SumMas2D(mas, 0, col);
  for (int i = 0; i < row; i++)
  {
    sum[i] = SumMas2D(mas, i, col);
    if (sum[i] > max)
    {
      max = sum[i];
    }
  }
  double numX = (double)max / height;
  double maxX = max - numX;
  for (int i = 0; i < height; i++)
  {
    for (int j = 0; j < row; j++)
    {
      if (sum[j] >= maxX) { cout << " + "; }
      else { cout << "   "; }
    }
    cout << endl;
    maxX -= numX;
  }
  for (int i = 0; i < row; i++)
  {
    cout << setw(2) << i + 1 << " ";
  }
  cout << endl;
}
