#include <iostream>
using namespace std;

/*
3. **ѕользователь вводит с клавиатуры врем€ начала
и врем€ завершени€ телефонного разговора
(часы, минуты и секунды).
ѕосчитать стоимость разговора, если стоимость
минуты - 30 копеек.

*/
int main()
{
  //setlocale(LC_ALL, "Rus"); // установка русской локализации в консоли
  system("chcp 1251 > nul"); // установка кодовой страницы 1251 с помощью консольной команды CHCP
  // 1) ¬вод исходных данных
  // int - им€ целого типа данных, целочисленный тип
  // int - Integer
  int Hour, Minuts, Second; // переменные дл€ часов, минут, секунд
  cout << "¬ведите врем€ начала разговора („ ћ —): "; // 1 23 14
  cin >> Hour >> Minuts >> Second;
  /* ¬вод данных времени (ч м с) по отдельности
  * внешне не будет разницы при
  * использовании cin >> Hour >> Minuts >> Second;
  cin >> Hour;
  cin >> Minuts;
  cin >> Second;
  */
  cout << "¬ведите врем€ конца разговора („ ћ —): ";
  int Hour2, Minuts2, Second2; // int H2, M2, S2;
  cin >> Hour2 >> Minuts2 >> Second2;
  // H1:M1:S1 --> H2:M2:S2 --> минут * 30

  // 2) ѕеревод времени в секунды
  // Time1 - врем€ начала разговора в секундах
  int Time1 = Hour * 3600 + Minuts * 60 + Second;
  // Time2 - врем€ окончани€ разговора в секундах
  int Time2 = Hour2 * 3600 + Minuts2 * 60 + Second2;

  // 3) ¬ычисление стоимости разговора
  int Dlit = Time2 - Time1; // длительность разговора в секундах
  Dlit = Dlit / 60; // перевод длительности в минуты
  // Dlit /= 60; - сокращенный вариант

  cout << "—тоимость разговора: " << Dlit * 30
    << " копеек" << endl;

  cout << "—тоимость разговора: " << Dlit * 30 / 100
    << " рублей" << endl;

  cout << "—тоимость разговора: "
    << (float)Dlit * 30 / 100 << " рублей" << endl;
  cout << "—тоимость разговора: "
    << (double)Dlit * 30 / 100 << " рублей" << endl;

  cout << "—тоимость разговора: "
    << Dlit * 30.0 / 100 << " рублей" << endl;

  //cin.get(); cin.get(); // блокировка консоли
  system("pause > nul"); // блокировка консоли с помощью консольной команды PAUSE
}
