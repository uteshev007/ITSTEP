#include <iostream>
using namespace std;

/*
2. *Пользователь вводит с клавиатуры расстояние
до аэропорта и время, за которое нужно доехать.
Вычислить скорость, с которой ему нужно ехать.
V = S / t

3. **Пользователь вводит с клавиатуры время начала
и время завершения телефонного разговора
(часы, минуты и секунды).
Посчитать стоимость разговора, если стоимость
минуты - 30 копеек.

*/
int main()
{
  //setlocale(LC_ALL, "Rus"); // установка русской локализации в консоли
  system("chcp 1251 > nul"); // установка кодовой страницы 1251 с помощью консольной команды CHCP
  // V = S / t
  cout << "Введите расстояние до аэропорта: ";
  // float - дробный, вещественный тип данных
  float S; // S - переменная для хранения расстояния, 
  // cin - Console INput - объект для ввода информации с консоли в программу
  cin >> S; // ввод значеия для расстояния, S
  cout << "Введите время в минутах: ";
  float t; // объявление переменной для хранения времени,
  cin >> t; // ввод времени с консоли в программу
  // 3) вычисление V
  //float V = S / t;
  float V; // объявление переменной для хранения скрости
  V = S / t;
  // Вывод результата для сорости
  cout << "Скорость должна быть: " << V
    << " км/мин" << endl;
  // перевод в км/ч
  // I вариант - вычисления во время вывода
  cout << "Скорость должна быть: " << V*60
    << " км/час" << endl;
  // II вариант
  float V2 = V * 60;
  cout << "Скорость должна быть: "<<V2<< " км/ч\n";
  // III вариант
  V *= 60; // сокращенная запись для V = V * 60;
  cout << "Скорость должна быть: "<<V<< " км/ч\n";

  // Без использования переменной V
  cout << "Скорость должна быть: " <<
    S / t * 60 << "км/ч" << endl;

  //cin.get(); cin.get(); // блокировка консоли
  system("pause > nul"); // блокировка консоли с помощью консольной команды PAUSE
}
