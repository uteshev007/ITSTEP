*реализовать перегруженные варианты функции (для типов int, long long, float, double);

int Max(int a, int b) {
  if(a > b){ return a; }
  else { return b; }
}

int Max(int a, int b, int c){
  int temp = Max(a,b);
  int res = Max(temp, c);
  return res;
  return Max( Max(a, b), c );
}
