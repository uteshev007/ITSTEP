#include <iostream>
#include <iomanip>
#include <time.h> // time()
using namespace std;

/*
3) **В двумерном массиве целых чисел посчитать
сумму элементов:
a) в каждой строке;
b) в каждом столбце;
c) ***одновременно по всем строкам и всем столбцам.
Оформить следующим образом:
  3  5  6  7 | 21
 12  1  1  1 | 15
  0  7 12  1 | 20
------------------
 15 13 19  9 | 56
*/

int main()
{
  setlocale(LC_ALL, "Rus");
  srand(time(0));
  const int kvart = 4;
  const int year = 3;
  int mas[year][kvart];
  int min, max;
  int total = 0;
  cout << "Введите минимальный и максимальные числа - ";
  cin >> min >> max;

  for (int i = 0; i < year; i++) {
    for (int j = 0; j < kvart; j++) {
      mas[i][j] = rand() % (max - min + 1) + min;
      total += mas[i][j];

    }
  }



  for (int i = 0; i < year; i++) {
    int summ = 0;
    for (int j = 0; j < kvart; j++) {
      cout << setw(3) << mas[i][j] << "   ";
      summ += mas[i][j];
    }
    cout << setw(3) << "| " << summ << endl;
  }
  for (int i = 0; i < kvart; i++) {
    cout << "------";
  }
  cout << "-+-----" << endl;
  for (int j = 0; j < kvart; j++) {
    int summ = 0;
    for (int i = 0; i < year; i++) {
      summ += mas[i][j];
    }
    cout << setw(3) << summ << "   ";
  }
  cout << " | " << total << "   ";

  cin.get(); cin.get();
}
