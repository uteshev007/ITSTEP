#include <iostream>
#include <iomanip>
#include <time.h> // time()
using namespace std;

/*
1) *ѕользователь вводит с клавиатуры числа N и M.
ќрганизуйте заполнение матрицы (двумерного массива)N x M
a) вводом с клавиатуры;
b) случайными значени€ми;
после чего выведите ее на экран.
*/
int main()
{
  setlocale(LC_ALL, "Rus");
  srand(time(0));
  // размеры двумерного массива
  const int N = 4; // кол-во строк в двумерном массиве
  const int M = 6; // кол-во столбцов в двумерном массиве
  //
  int mas[N][M]; // создание, объ€вление двумерного массива
  // размером N x M, [N,M]

  // иниц-ци€ массива списком, списочна€ иниц-ци€
  int mas2[N][M] = {
    {1,2,3,4,5}, // [0] иниц-ци€ первой строки двум.массива 
    {10,20,},    // [1]
    {},          // [2] иниц-ци€ строки нул€ми
    {-1,-3,-2, -8,45} // [3]
  };
  // иниц-ци€ случ. значени€ми
  for (int i = 0; i < N; i++) {
    // цикл дл€ работы с одной (i-ой) строкой
    for (int j = 0; j < M; j++) {
      mas[i][j] = rand() % 20 + 1; // [1;20]
      cout << setw(2) << mas[i][j] << " ";
    }
    cout << endl; // конец строки - дл€ вывода на экран
  }
  cout << endl; // разделитель
  // цикл дл€ печати массива mas2
  for (int i = 0; i < N; i++) { // по строкам
    // печать одной строки
    for (int j = 0; j < M; j++) { // по столбцам
      cout << setw(3) << mas2[i][j] << " "; // вывод элемента из i-ой строки и j-ой позиции
    }
    cout << endl; // конец вывода i-ой строки
  }

  cin.get(); cin.get();
}
