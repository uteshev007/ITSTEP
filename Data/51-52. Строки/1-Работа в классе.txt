#define SIZE 1000
int Countwork1(char str[]) {
  if (str[0] == '\0') {return 0;}
  int count = 0;
  for (int i = 0; str[i] != '\0'; i++) {
    if (str[i] == ' ') {
      count++;
    }
  }
  return count+1;
}
