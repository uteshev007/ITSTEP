#define SIZE 1000
int Countwork1(char str[]) {
  if (str[0] == '\0') { return 0; }
  int count = 0;
  for (int i = 0; str[i] != '\0'; i++) {
    if (str[i] == ' ') {
      count++;
    }
  }
  return count + 1;
}

int Coutword2(char str[])
{
  //int gap = 0;

  //else {gap = 1;}
  if (str[0] == '\0'){ return 0;}
  int gap = (str[0] == ' ') ? 0 : 1; //тернарный оператор
   
  for (int i = 0; str[i] != '\0'; i++)
  {
    if (str[i] == ' ' && str[i+1] != ' ' &&
        str[i+1] !='\0')
    {
      gap++;
    }
  }
  return gap;
}
