#include <iostream>
using namespace std;

/*
18. ***Пользователь вводит с клавиатуры объем
флешки в гигабайтах.
Посчитать, сколько на нее поместится
- фильмов размером 760 мб,
- музыкальных клипов объемом 95 мб,
- музыкальных композиций объемом 7 мб,
- текстовых документов объемом 350 кб.
а) пересчитать результат, учитывая, что сначала
пользователь записывает фильмы, пока для них есть
место, после того, как фильмы уже не влазят,
но осталось еще место записывает клипы,
после того, как ни один клип уже не помещается
на флешку, на оставшееся место записывает музыку, 
и свободное место заполняет текстовыми документами.
*/

int main()
{
  setlocale(LC_ALL, "Rus");
  double FlashSize; // переменная для полного размера флешки в Гб
  cout << "Введите размер флешки: ";
  cin >> FlashSize; // 2.5  (Гб)
  // Гб ==> Кб
  int Flashka = FlashSize * 1024 * 1024; // Gb ==> Kb
  cout << "Размер флешки " << Flashka << " Кб\n";
  // костантные переменные для размеров файлов: фильмы, клипы и т.д.
  const int FilmSize = 760 * 1024; // 760 Mb => Kb
  //FilmSize = 0; - ошибка, т.к. переменная FilmSize объявлена, как константная переменная
  const int ClipSize = 95 * 1024; // 95Mb => Kb
  const int MuzSize = 7 * 1024;   // 7Mb => Kb
  const int TextSize = 350; // Kb
  // 1) расчет и вывод результатов
  cout << "Фильмов: " << Flashka / FilmSize
    << " файлов\n";
  cout << "Клипов: " << Flashka / ClipSize << " файлов\n";
  cout << "Муз. файлов: "<< Flashka / MuzSize<<endl;
  cout << "Текстовых документов: "
    << Flashka / TextSize << " файлов" << endl;
  // 2) дозапись на флешку
  cout << endl << "С дозаписью файлов на флешку" << endl;
  cout<<"Фильмов: "<<Flashka/FilmSize<<" файлов\n";
  // Flashka = Flashka % FilmSize;
  Flashka %= FilmSize; // остаток памяти на флешке после записи фильмов
  cout<<"Клипов: "<<Flashka / ClipSize<<" файлов\n";
  Flashka %= ClipSize; // остаток памяти после клипов
  cout << "Муз.файлов: " << Flashka / MuzSize << endl;
  Flashka %= MuzSize; // остаток памяти после муз. файлов
  cout << "Документов: " << Flashka / TextSize
    << " файлов" << endl;
  Flashka %= TextSize; // остаток памяти после копирования документов
  cout << "Остаток памятим на флешке: "
    << Flashka << " Кбайт" << endl;

  cin.get(); cin.get();
}
