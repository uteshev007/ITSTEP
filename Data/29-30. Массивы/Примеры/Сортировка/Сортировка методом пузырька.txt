#include<iostream>
using namespace std;

int main() {
	const int size = 3000;
	int mas[size];

	for (int i = 0; i < size; i++) {
		mas[i] = rand();
	}
	//массив до сортировки
	//for (int i = 0; i < size; i++) {
	//	cout << mas[i] << " ";
	//}
	cout << "!!!!!";
	//сортировка пузырьком
	for (int j = 0; j < size - 1; j++) {
		for (int i = 0; i < size - 1 - j; i++) {
			if (mas[i] > mas[i + 1]) {
				int tmp = mas[i];
				mas[i] = mas[i + 1];
				mas[i + 1] = tmp;
			}
		}
	}
	cout << "!!!!!";
	cout << endl;
	// массив после сортировки
	/*for (int i = 0; i < size; i++) {
